Let

Let (in contrast with var) is block scoped.

let snack = 'Meow Mix';

function getFood(food) {
    if (food) {
        let snack = 'Friskies';
    }
    return snack;
}

getFood(true); // 'Meow Mix'

const MY_CONST = 23;
MY_CONST = 12; // Error

let MY_CONST = 12; // Error
